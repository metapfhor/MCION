1function error = naivePerfusionSSEP2X4Pulse(ton,toff,Ttot,T,I)
  2global A J p0 err IVMon IVMoff;
  3    %#codgen;
  4    p0=zeros(1,17);
  5    p0(1)=1;
  6    ton=[ton,Ttot];
  7    err=0;
  8    A0=A;
  9    J0=J;
 10    J=0;
 11    
 12    ind=find(ton<IVMon);
 13    ion=ind(end);
 14    ind=find(ton<IVMoff);
 15    ioff=ind(end);
 16    
 17    for i=1:ion-1
 18        indOn=T>ton(i)&T<=toff(i);
 19        indOff=T>toff(i)&T<=ton(i+1);
 20        
 21        A=A0;
 22        linearODESSEP2X4(T(indOn)-ton(i),I(indOn));
 23        
 24        A=0;
 25        linearODESSEP2X4(T(indOff)-toff(i),I(indOff));
 26    end
 27    
 28        indOn=T>ton(ion)&T<=toff(ion);
 29        indOff=T>toff(ion)&T<=IVMon;
 30        
 31        A=A0;
 32        linearODESSEP2X4(T(indOn)-ton(ion),I(indOn));
 33        
 34        A=0;
 35        linearODESSEP2X4(T(indOff)-toff(ion),I(indOff));
 36        
 37        J=J0;
 38            
 39        indPre=T>IVMon&T<=ton(ion+1);
 40        indOn=T>ton(ion+1)&T<=toff(ion+1);
 41        indOff=T>toff(ion+1)&T<=ton(ion+2);
 42        
 43        linearODESSEP2X4(T(indPre)-IVMon,I(indPre));
 44        
 45        A=A0;
 46        linearODESSEP2X4(T(indOn)-ton(ion+1),I(indOn));
 47        
 48        A=0;
 49        linearODESSEP2X4(T(indOff)-toff(ion+1),I(indOff));
 50        
 51        
 52    
 53    for i=ion+2:ioff-1
 54        indOn=T>ton(i)&T<=toff(i);
 55        indOff=T>toff(i)&T<=ton(i+1);
 56        
 57        A=A0;
 58        linearODESSEP2X4(T(indOn)-ton(i),I(indOn));
 59        
 60        A=0;
 61        linearODESSEP2X4(T(indOff)-toff(i),I(indOff));
 62    end
 63    if (ioff>length(toff))
 64        error=err;
 65        return;
 66    end
 67    
 68        indOn=T>ton(ioff)&T<=toff(ioff);
 69        indOff=T>toff(ioff)&T<=IVMoff;
 70        
 71        A=A0;
 72        linearODESSEP2X4(T(indOn)-ton(ioff),I(indOn));
 73        
 74        A=0;
 75        linearODESSEP2X4(T(indOff)-toff(ioff),I(indOff));
 76        
 77        J=0;
 78        
 79        indPre=T>IVMoff&T<=ton(ioff+1);
 80        indOn=T>ton(ioff+1)&T<=toff(ioff+1);
 81        indOff=T>toff(ioff+1)&T<=ton(ioff+2);
 82        
 83        linearODESSEP2X4(T(indPre)-IVMoff,I(indPre));
 84        
 85        A=A0;
 86        linearODESSEP2X4(T(indOn)-ton(ioff+1),I(indOn));
 87        
 88        A=0;
 89        linearODESSEP2X4(T(indOff)-toff(ioff+1),I(indOff));
 90        
 91    for i=ioff+1:length(ton)-1
 92        indOn=T>ton(i)&T<=toff(i);
 93        indOff=T>toff(i)&T<=ton(i+1);
 94
 95        A=A0;
 96        linearODESSEP2X4(T(indOn)-ton(i),I(indOn));
 97
 98        A=0;
 99        linearODESSEP2X4(T(indOff)-toff(i),I(indOff));
100    end
101    
102    error=err;
103end
