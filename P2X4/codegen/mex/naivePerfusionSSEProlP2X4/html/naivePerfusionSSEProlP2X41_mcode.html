1function error = naivePerfusionSSEProlP2X4(ton,toff,Ttot,T,I)
 2global A J p0 err IVMon IVMoff Acell;
 3    %This is the function which the sum of squared errors for the Pulse
 4    %protocol is calculated. The globals are declared above so they can be
 5    %used in this function.
 6    %The variable p0 is initialized here.

12    %The variable 'err' is initialized at 0.

17    A0=A;
18    J0=J;
19    
20        J=J0;
21            
22        indPre=T>IVMon&T<=ton(1);
23        indOn=T>ton(1)&T<=toff(1);

25        
26        A=0;
27        linearODESSEP2X4(T(indPre)-IVMon,I(indPre));
28        
29        A=A0;
30        linearODESSEP2X4(T(indOn)-ton(1),I(indOn));
31        
32        A=0;
33        linearODESSEP2X4(T(indOff)-toff(1),I(indOff));
34        
35        indOn2=T>ton(2)&T<=toff(2);

37

39        
40        linearODESSEP2X4(T(indOn2)-ton(2),I(indOn2));
41        
42        A=0;
43        
44        linearODESSEP2X4(T(indOff2)-toff(2),I(indOff2));

47end
