1function error = naivePerfusionSSEPulseP2X4(ton,toff,Ttot,T,I)
  2global A J p0 err IVMon IVMoff Acell;
  3    %This is the function which the sum of squared errors for the Pulse
  4    %protocol is calculated. The globals are declared above so they can be
  5    %used in this function.
  6    %The variable p0 is initialized here.
  7    p0=zeros(1,9);
  8    p0(1)=1;
  9    %The matrix ton contains the elements ton and Ttot. MAY WANT TO CHECK
 10    %THIS.
 11    ton=[ton,Ttot];
 12    %The variable 'err' is initialized at 0.
 13    err=0;
 14    %A0 is made equal to the concentration of ATP that is applied and
 15    %declared in the globals of the experimental array. The same thing
 16    %applies to J0, which corresponds to the concentration of IVM.
 17    A0=A;
 18    J0=J;
 19    %First the concentration of IVM is made zero.
 20    J=0;
 21    
 22    ind=find(ton<IVMon);
 23    ion=ind(end);
 24    ind=find(ton<IVMoff);
 25    ioff=ind(end);
 26    
 27    for i=1:ion-1
 28        indOn=T>ton(i)&T<=toff(i);
 29        indOff=T>toff(i)&T<=ton(i+1);
 30        
 31        A=A0;
 32        linearODESSEP2X4(T(indOn)-ton(i),I(indOn));
 33        
 34        A=0;
 35        linearODESSEP2X4(T(indOff)-toff(i),I(indOff));
 36    end
 37    
 38        indOn=T>ton(ion)&T<=toff(ion);
 39        indOff=T>toff(ion)&T<=IVMon;
 40        
 41        A=A0;
 42        linearODESSEP2X4(T(indOn)-ton(ion),I(indOn));
 43        
 44        A=0;
 45        linearODESSEP2X4(T(indOff)-toff(ion),I(indOff));
 46        
 47        J=J0;
 48            
 49        indPre=T>IVMon&T<=ton(ion+1);
 50        indOn=T>ton(ion+1)&T<=toff(ion+1);
 51        indOff=T>toff(ion+1)&T<=ton(ion+2);
 52        
 53        linearODESSEP2X4(T(indPre)-IVMon,I(indPre));
 54        
 55        A=A0;
 56        linearODESSEP2X4(T(indOn)-ton(ion+1),I(indOn));
 57        
 58        A=0;
 59        linearODESSEP2X4(T(indOff)-toff(ion+1),I(indOff));
 60        
 61        
 62    
 63    for i=ion+2:ioff-1
 64        indOn=T>ton(i)&T<=toff(i);
 65        indOff=T>toff(i)&T<=ton(i+1);
 66        
 67        A=A0;
 68        linearODESSEP2X4(T(indOn)-ton(i),I(indOn));
 69        
 70        A=0;
 71        linearODESSEP2X4(T(indOff)-toff(i),I(indOff));
 72    end
 73    if (ioff>length(toff))
 74        error=err;
 75        return;
 76    end
 77    
 78        indOn=T>ton(ioff)&T<=toff(ioff);
 79        indOff=T>toff(ioff)&T<=IVMoff;
 80        
 81        A=A0;
 82        linearODESSEP2X4(T(indOn)-ton(ioff),I(indOn));
 83        
 84        A=0;
 85        linearODESSEP2X4(T(indOff)-toff(ioff),I(indOff));
 86        
 87        J=0;
 88        
 89        if length(toff)<=ioff
 90            error=err;
 91            return
 92        end
 93        
 94        indPre=T>IVMoff&T<=ton(ioff+1);
 95        indOn=T>ton(ioff+1)&T<=toff(ioff+1);
 96        indOff=T>toff(ioff+1)&T<=ton(ioff+2);
 97        
 98        linearODESSEP2X4(T(indPre)-IVMoff,I(indPre));
 99        
100        A=A0;
101        linearODESSEP2X4(T(indOn)-ton(ioff+1),I(indOn));
102        
103        A=0;
104        linearODESSEP2X4(T(indOff)-toff(ioff+1),I(indOff));
105        
106    for i=ioff+1:length(ton)-1
107        indOn=T>ton(i)&T<=toff(i);
108        indOff=T>toff(i)&T<=ton(i+1);
109
110        A=A0;
111        linearODESSEP2X4(T(indOn)-ton(i),I(indOn));
112
113        A=0;
114        linearODESSEP2X4(T(indOff)-toff(i),I(indOff));
115    end
116    
117    error=err;
118end
